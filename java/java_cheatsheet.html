Class Anatomy
public class BankAccount {
  // fields (state)
  private String owner;
  private double balance;
  private static int totalAccounts = 0; // class-level

  // constructor
  public BankAccount(String owner, double balance) {
    this.owner = owner;
    this.balance = balance;
    totalAccounts++;
  }

  // copy constructor
  public BankAccount(BankAccount other) {
    this(other.owner, other.balance);
  }

  // getter / setter (encapsulation)
  public double getBalance() { return balance; }
  public void deposit(double amount) {
    if (amount > 0) balance += amount;
  }

  // static method
  public static int getTotalAccounts() { return totalAccounts; }

  // overriding Object methods
  @Override
  public String toString() {
    return String.format("Account[%s, %.2f]", owner, balance);
  }
}

extends & super
class Animal {
  String name;
  Animal(String name) { this.name = name; }
  void speak() { System.out.println("..."); }
}

class Dog extends Animal {
  String breed;

  Dog(String name, String breed) {
    super(name);      // must be first statement
    this.breed = breed;
  }

  @Override
  void speak() {        // runtime polymorphism
    System.out.println(name + ": Woof!");
  }

  void fetch() {
    super.speak();    // call parent method
    System.out.println("Fetching...");
  }
}

// Usage
Animal a = new Dog("Rex", "Lab"); // upcasting
a.speak();                             // → "Rex: Woof!" (dynamic dispatch)
Dog d = (Dog) a;                       // downcasting
if (a instanceof Dog dog) dog.fetch(); // Java 16+ pattern match
Abstract Classes
abstract class Shape {
  String color;

  Shape(String color) { this.color = color; }

  abstract double area();         // must override
  abstract double perimeter();

  void describe() {               // concrete method
    System.out.printf("%s: area=%.2f%n", color, area());
  }
}

class Circle extends Shape {
  double r;
  Circle(double r) { super("red"); this.r = r; }

  @Override double area()      { return Math.PI * r * r; }
  @Override double perimeter() { return 2 * Math.PI * r; }
}
Object class — Key Methods to Override
class Point {
  int x, y;
  Point(int x, int y) { this.x=x; this.y=y; }

  @Override
  public boolean equals(Object o) {
    if (this == o) return true;
    if (!(o instanceof Point p)) return false;
    return x == p.x && y == p.y;
  }

  @Override
  public int hashCode() { return Objects.hash(x, y); }

  @Override
  public String toString() { return "(" + x + "," + y + ")"; }
}
Interface Basics & Default Methods Java 8+
interface Drawable {
  void draw();                         // abstract (implicit)

  default void print() {              // default implementation
    System.out.println("Drawing: " + getClass().getSimpleName());
  }

  static Drawable ofCircle() {        // static factory
    return () -> System.out.println("○");

}

interface Resizable { void resize(double factor); }

// Multiple interface implementation (no diamond problem)
class Square implements Drawable, Resizable {
  double side;
  @Override public void draw()               { System.out.println("□"); }
  @Override public void resize(double f)    { side *= f; }
}

Generic Classes & Methods
// Generic class
class Pair<A, B> {
  private final A first;
  private final B second;

  public Pair(A first, B second) { this.first=first; this.second=second; }
  public A getFirst()  { return first; }
  public B getSecond() { return second; }

  public static <A,B> Pair<A,B> of(A a, B b) { return new Pair<>(a, b); }
}

// Generic method
public static <T extends Comparable<T>> T max(T a, T b) {
  return a.compareTo(b) >= 0 ? a : b;
}

max(3, 7)       // → 7
max("apple", "mango") // → "mango"
Wildcards
// Unbounded: accepts any type
void printList(List<?> list) { list.forEach(System.out::println); }

// Upper bounded: T or subtype (read-only pattern)
double sum(List<? extends Number> nums) {
  return nums.stream().mapToDouble(Number::doubleValue).sum();
}

// Lower bounded: T or supertype (write pattern)
void addNumbers(List<? super Integer> list) {
  list.add(1); list.add(2);
}

// Lazy intermediate ops + eager terminal op
List<String> result = names.stream()
  .filter(s -> s.startsWith("A"))      // Predicate
  .map(String::toUpperCase)              // Function
  .sorted()                              // natural order
  .distinct()                            // remove dups
  .limit(5)                             // max 5 elements
  .collect(Collectors.toList());
